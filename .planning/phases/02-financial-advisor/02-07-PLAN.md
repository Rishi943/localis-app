### 8. Fix build_finance_context — replace period_label with strftime

Find `build_finance_context(` function (around line 690). Replace all `period_label` column references with `strftime('%Y-%m', date)` date-derived filtering:

```python
def build_finance_context(period: str = 'all') -> str:
    conn = database._connect_db()
    c = conn.cursor()

    period_filter = period if period and period not in ('all', 'All time') else None
    period_clause = "AND strftime('%Y-%m', date) = ?" if period_filter else ""
    period_params = [period_filter] if period_filter else []

    c.execute(f"""
        SELECT category, SUM(amount) as total, COUNT(*) as count
        FROM fin_transactions
        WHERE type = 'debit' {period_clause}
        GROUP BY category ORDER BY total DESC
    """, period_params)
    cat_totals = c.fetchall()

    c.execute(f"""
        SELECT description, SUM(amount) as total
        FROM fin_transactions
        WHERE type = 'debit' {period_clause}
        GROUP BY description ORDER BY total DESC LIMIT 5
    """, period_params)
    top_merchants = c.fetchall()

    c.execute("SELECT budgets, goal_type, horizon FROM fin_goals ORDER BY created_at DESC LIMIT 1")
    goals_row = c.fetchone()
    budgets = {}
    goal_type = ''
    horizon = ''
    if goals_row:
        try: budgets = json.loads(goals_row[0] or '{}')
        except: pass
        goal_type = goals_row[1] or ''
        horizon = goals_row[2] or ''

    conn.close()

    period_display = period if period not in ('all', 'All time') else 'All time'
    lines = [
        "FINANCIAL CONTEXT (from user's bank data):",
        f"Period: {period_display}",
        "",
        "Spend by category:",
    ]
    for cat, total, count in cat_totals:
        budget = budgets.get(cat)
        budget_note = f" (budget: ${budget:.0f})" if budget else ""
        lines.append(f"  {cat}: ${total:.2f} ({count} transactions){budget_note}")

    if top_merchants:
        lines.append("")
        lines.append("Top merchants:")
        for desc, total in top_merchants:
            lines.append(f"  {desc}: ${total:.2f}")

    if goal_type:
        lines.append("")
        lines.append(f"User's financial goal: {goal_type}")
    if horizon:
        lines.append(f"Time horizon: {horizon}")

    return "\n".join(lines)
```

function renderBudgetActual(categories, budgets) {
    const container = document.getElementById('fin-budget-chart');
    if (!container) return;

    const ALL_CATEGORIES = ['Food', 'Transport', 'Shopping', 'Utilities', 'Entertainment', 'Health & Fitness', 'Government & Fees', 'Other'];
    const spendMap = {};
    (categories || []).forEach(c => { spendMap[c.category] = c.total; });

    const rows = ALL_CATEGORIES.map(cat => {
        const actual = spendMap[cat] || 0;

               : 'fin-bar-fill')

            <div class="fin-bar-track">
                <div class="${barClass}" style="--pct:${pct}"></div>
            </div>
            <span class="fin-bar-amount">$${actual.toFixed(2)}${hasBudget ? ' / $' + budget.toFixed(0) : ''}</span>

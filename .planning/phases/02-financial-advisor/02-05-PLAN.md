```javascript
async function _startOnboarding() {
    const history = document.getElementById('fin-onboarding-history');
    const input = document.getElementById('fin-onboarding-input');
    const sendBtn = document.getElementById('fin-onboarding-send');
    const skipBtn = document.getElementById('fin-onboarding-skip');
    if (!history || !input) return;

    const ALL_CATEGORIES = ['Food', 'Transport', 'Shopping', 'Utilities', 'Entertainment', 'Health & Fitness', 'Government & Fees', 'Other'];
    let stepIndex = 0;
    const answers = { goal_type: '', life_events: '', budgets: {}, horizon: '' };

    const STEPS = [
        {
            key: 'welcome',
            message: "Let's talk about your finances. What are you working toward?",
            isWelcome: true, // auto-advance, no input needed
        },
        {
            key: 'goal_type',
            message: "What's your main goal right now — saving more, investing, or working toward something specific like a trip or a big purchase?",
        },
        {
            key: 'life_events',
            message: "Any life events you're planning toward? Vacation, wedding, buying a home, building an emergency fund — anything like that?",
        },
        {
            key: 'budgets',
            message: "Want to set monthly budgets for your spending categories? Enter 0 for any you'd rather not track.",
            isForm: true,
        },
        {
            key: 'horizon',
            message: "What's your time horizon — are you thinking weeks, months, or a few years out?",
        },
    ];

    function _appendAssistant(text) {
        const div = document.createElement('div');
        div.className = 'msg msg-assistant';
        div.innerHTML = `<div class="msg-text">${text}</div>`;
        history.appendChild(div);
        history.scrollTop = history.scrollHeight;
    }

    function _appendUser(text) {
        const div = document.createElement('div');
        div.className = 'msg msg-user';
        div.innerHTML = `<div class="msg-text">${text}</div>`;
        history.appendChild(div);
        history.scrollTop = history.scrollHeight;
    }

    function _showBudgetForm() {
        // Hide text input row
        const inputRow = input.closest('.fin-onboarding-input-row') || input.closest('.fin-chat-input-row');
        if (inputRow) inputRow.style.display = 'none';

        const card = document.createElement('div');
        card.className = 'fin-onboarding-budget-card';

        let gridHTML = '<div class="fin-budget-form-grid">';
        ALL_CATEGORIES.forEach(cat => {
            gridHTML += `<div class="fin-budget-form-row">
                <span class="fin-budget-form-label">${cat}</span>
                <span class="fin-budget-form-dollar">$</span>
                <input type="number" class="fin-budget-form-input" data-category="${cat}" placeholder="0" min="0" step="50">
            </div>`;
        });
        gridHTML += '</div>';
        gridHTML += `<button class="fin-budget-form-submit">Set budgets</button>`;

        card.innerHTML = gridHTML;
        history.appendChild(card);
        history.scrollTop = history.scrollHeight;

        card.querySelector('.fin-budget-form-submit').addEventListener('click', () => {
            const budgets = {};
            card.querySelectorAll('.fin-budget-form-input').forEach(inp => {
                const cat = inp.dataset.category;
                const val = parseFloat(inp.value) || 0;
                if (val > 0) budgets[cat] = val;
            });
            answers.budgets = budgets;

            // Disable form
            card.querySelectorAll('input, button').forEach(el => el.disabled = true);

            // Restore input row
            if (inputRow) inputRow.style.display = '';

            // Show user summary
            const summary = Object.entries(budgets).map(([k, v]) => `${k}: $${v}`).join(', ') || 'No budgets set';
            _appendUser(summary);

            stepIndex++;
            _advanceStep();
        });
    }

    function _advanceStep() {
        if (stepIndex >= STEPS.length) {
            _completeOnboarding();
            return;
        }

        const step = STEPS[stepIndex];
        _appendAssistant(step.message);

        if (step.isForm) {
            _showBudgetForm();
        }
    }

    async function _completeOnboarding() {
        _appendAssistant("Got it. Head to your Dashboard when you're ready to upload your first bank statement.");

        // Save goals to backend
        try {
            await fetch('/finance/goals', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(answers),
            });
        } catch (e) {
            console.warn('[Finance] Failed to save goals:', e);
        }

        // Wait 1.5s then switch to dashboard
        setTimeout(() => {
            document.querySelectorAll('.fin-tab').forEach(t => t.style.display = '');
            _activateTab('dashboard');
            _refreshDashboard();
        }, 1500);
    }

    // Wire send button and Enter key for text steps
    function _onSend() {
        const text = input.value.trim();
        if (!text) return;

        const step = STEPS[stepIndex];
        if (step.isForm) return; // Form handles its own submit

        _appendUser(text);
        answers[step.key] = text;
        input.value = '';

        stepIndex++;
        _advanceStep();
    }

    sendBtn?.addEventListener('click', _onSend);
    input?.addEventListener('keydown', (e) => {
        if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); _onSend(); }
    });

    // Wire skip button
    skipBtn?.addEventListener('click', () => {
        document.querySelectorAll('.fin-tab').forEach(t => t.style.display = '');
        _activateTab('dashboard');
        _refreshDashboard();
    });

    // Start: show welcome then auto-advance to first question
    _appendAssistant(STEPS[0].message);
    stepIndex = 1;
    _advanceStep();
}
```

```javascript
async function _sendFinanceChat() {
    const input = document.getElementById('fin-chat-input');
    const sendBtn = document.getElementById('fin-chat-send');
    const history = document.getElementById('fin-chat-history');
    const emptyState = document.getElementById('fin-chat-empty');
    if (!input || !history) return;

    const text = input.value.trim();
    if (!text) return;

    emptyState?.classList.add('hidden');
    input.value = '';
    sendBtn.disabled = true;

    // Append user message
    const userDiv = document.createElement('div');
    userDiv.className = 'msg msg-user';
    userDiv.innerHTML = `<div class="msg-text">${text}</div>`;
    history.appendChild(userDiv);

    // Add to in-memory history
    _finChatHistory.push({ role: 'user', content: text });

    // Create assistant message container
    const assistDiv = document.createElement('div');
    assistDiv.className = 'msg msg-assistant';
    const textEl = document.createElement('div');
    textEl.className = 'msg-text';
    assistDiv.appendChild(textEl);
    history.appendChild(assistDiv);
    history.scrollTop = history.scrollHeight;

    let buffer = '';
    let rafScheduled = false;

    try {
        const res = await fetch('/finance/chat', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                message: text,
                history: _finChatHistory.slice(-10), // last 10 messages for context
                period: _currentPeriod,
            }),
        });

        if (!res.ok) {
            const err = await res.json().catch(() => ({}));
            textEl.textContent = err.detail || "I don't have any transaction data for that period. Upload a CSV for that time range first.";
            sendBtn.disabled = false;
            return;
        }

        const reader = res.body.getReader();
        const decoder = new TextDecoder();

        while (true) {
            const { done, value } = await reader.read();
            if (done) break;

            const chunk = decoder.decode(value, { stream: true });
            // Parse SSE data lines
            const lines = chunk.split('\n');
            for (const line of lines) {
                if (line.startsWith('data: ')) {
                    const payload = line.slice(6);
                    if (payload === '[DONE]') continue;
                    try {
                        const obj = JSON.parse(payload);
                        const token = obj.choices?.[0]?.delta?.content || obj.token || '';
                        if (token) {
                            buffer += token;
                            if (!rafScheduled) {
                                rafScheduled = true;
                                requestAnimationFrame(() => {
                                    textEl.textContent = buffer;
                                    history.scrollTop = history.scrollHeight;
                                    rafScheduled = false;
                                });
                            }
                        }
                    } catch (e) {
                        // Non-JSON data line (e.g. plain token)
                        buffer += payload;
                        if (!rafScheduled) {
                            rafScheduled = true;
                            requestAnimationFrame(() => {
                                textEl.textContent = buffer;
                                history.scrollTop = history.scrollHeight;
                                rafScheduled = false;
                            });
                        }
                    }
                }
            }
        }

        // Final render with markdown if buildMessageHTML exists
        if (buffer && typeof buildMessageHTML === 'function') {
            textEl.innerHTML = buildMessageHTML(buffer);
        } else {
            textEl.textContent = buffer;
        }

        _finChatHistory.push({ role: 'assistant', content: buffer });

    } catch (e) {
        textEl.textContent = 'Something went wrong. Try again.';
        console.warn('[Finance] Chat error:', e);
    }

    sendBtn.disabled = false;
    history.scrollTop = history.scrollHeight;
}
```

```python
def build_finance_context(period: str = 'all') -> str:
    conn = database._connect_db()
    c = conn.cursor()

    period_clause = ""
    params = []
    if period and period != 'all':
        period_clause = "AND strftime('%Y-%m', date) = ?"
        params = [period]

    # Category totals (debits only)
    c.execute(f"""
        SELECT category, SUM(amount) as total, COUNT(*) as count
        FROM fin_transactions
        WHERE type = 'debit' {period_clause}
        GROUP BY category ORDER BY total DESC
    """, params)
    cat_totals = c.fetchall()

    # Top 5 merchants
    c.execute(f"""
        SELECT description, SUM(amount) as total, COUNT(*) as count
        FROM fin_transactions
        WHERE type = 'debit' {period_clause}
        GROUP BY description ORDER BY total DESC LIMIT 5
    """, params)
    top_merchants = c.fetchall()

    # Budget info
    c.execute("SELECT budgets, goal_type, horizon FROM fin_goals ORDER BY created_at DESC LIMIT 1")
    goals_row = c.fetchone()
    budgets = {}
    goal_type = ''
    horizon = ''
    if goals_row:
        try: budgets = json.loads(goals_row[0] or '{}')
        except: pass
        goal_type = goals_row[1] or ''
        horizon = goals_row[2] or ''

    conn.close()

    period_label = period if period != 'all' else 'All time'
    lines = [
        "FINANCIAL CONTEXT (from user's bank data):",
        f"Period: {period_label}",
        "",
        "Spend by category:",
    ]
    for cat, total, count in cat_totals:
        budget = budgets.get(cat)
        budget_note = f" (budget: ${budget:.0f})" if budget else ""
        lines.append(f"  {cat}: ${total:.2f} ({count} transactions){budget_note}")

    if top_merchants:
        lines.append("")
        lines.append("Top merchants:")
        for desc, total, count in top_merchants:
            lines.append(f"  {desc}: ${total:.2f} ({count}x)")

    if goal_type:
        lines.append("")
        lines.append(f"User's financial goal: {goal_type}")
    if horizon:
        lines.append(f"Time horizon: {horizon}")

    return "\n".join(lines)
```

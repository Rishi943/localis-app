const financeUI = (function() {
    let _open = false;
    let _currentPeriod = 'all';
    let _lineChart = null;   // Chart.js instance
    let _donutChart = null;  // Chart.js instance
    let _finChatHistory = []; // in-memory finance chat history

    // ---- Panel open/close ----
    function open() {
        _open = true;
        document.getElementById('finance-panel').classList.remove('hidden');
        _checkOnboarding();
        _loadPeriods();
        _escListener = _onEsc.bind(null);
        document.addEventListener('keydown', _escListener);
    }
    function close() {
        _open = false;
        document.getElementById('finance-panel').classList.add('hidden');
        if (_escListener) document.removeEventListener('keydown', _escListener);
    }
    let _escListener = null;
    function _onEsc(e) { if (e.key === 'Escape') close(); }

    // ---- Tab switching ----
    function _activateTab(tabName) {
        document.querySelectorAll('.fin-tab').forEach(t => {
            t.classList.toggle('active', t.dataset.tab === tabName);
            t.setAttribute('aria-selected', t.dataset.tab === tabName ? 'true' : 'false');
        });
        document.querySelectorAll('.fin-pane').forEach(p => p.classList.remove('active'));
        const paneId = tabName === 'dashboard' ? 'fin-pane-dashboard'
                     : tabName === 'chat' ? 'fin-pane-chat'
                     : 'fin-pane-onboarding';
        document.getElementById(paneId)?.classList.add('active');

        // Hide budget sidebar in Chat tab
        const sidebar = document.querySelector('.fin-budget-sidebar');
        if (sidebar) sidebar.style.display = tabName === 'chat' ? 'none' : '';
    }

    // ---- Period selector ----
    async function _loadPeriods() {
        try {
            const res = await fetch('/finance/periods');
            const data = await res.json();
            const select = document.getElementById('fin-period-select');
            const current = select.value;
            // Clear all options except "All time"
            select.innerHTML = '<option value="all">All time</option>';
            (data.periods || []).forEach(p => {
                // p is "YYYY-MM" string, format as "Mar 2026"
                const [y, m] = p.split('-');
                const dt = new Date(parseInt(y), parseInt(m) - 1);
                const label = dt.toLocaleDateString('en-US', { month: 'short', year: 'numeric' });
                const opt = document.createElement('option');
                opt.value = p;
                opt.textContent = label;
                select.appendChild(opt);
            });
            // Restore selection if it still exists
            if ([...select.options].some(o => o.value === current)) {
                select.value = current;
            }
        } catch (e) { console.warn('[Finance] Failed to load periods:', e); }
    }

    function _onPeriodChange() {
        _currentPeriod = document.getElementById('fin-period-select').value;
        _refreshDashboard();
    }

    // ---- Refresh ----
    async function _refreshDashboard() {
        const btn = document.getElementById('fin-refresh-btn');
        btn?.classList.add('spinning');
        btn && (btn.disabled = true);
        try {
            await _loadPeriods();
            await _loadDashboard(_currentPeriod);
        } finally {
            btn?.classList.remove('spinning');
            btn && (btn.disabled = false);
        }
    }

    // ---- Onboarding check ----
    async function _checkOnboarding() {
        try {
            const res = await fetch('/finance/status');
            const data = await res.json();
            if (!data.onboarding_done) {
                _activateTab('onboarding');
                // Hide regular tabs during onboarding
                document.querySelectorAll('.fin-tab').forEach(t => t.style.display = 'none');
                _startOnboarding();
            } else {
                document.querySelectorAll('.fin-tab').forEach(t => t.style.display = '');
                _activateTab('dashboard');
                _refreshDashboard();
            }
        } catch (e) {
            // Default to dashboard if status check fails
            document.querySelectorAll('.fin-tab').forEach(t => t.style.display = '');
            _activateTab('dashboard');
        }
    }

    function _initUpload() {
        const trigger = document.getElementById('fin-upload-trigger');

        const accountInput = document.getElementById('fin-account-label-input');

        trigger?.addEventListener('click', async () => {
            panel?.classList.remove('hidden');
            trigger.style.display = 'none';
            // Populate account datalist
            try {
                const res = await fetch('/finance/accounts');
                const data = await res.json();
                const dl = document.getElementById('fin-account-suggestions');
                dl.innerHTML = '';
                (data.accounts || []).forEach(a => {
                    const opt = document.createElement('option');
                    opt.value = a;
                    dl.appendChild(opt);
                });
            } catch (e) {}
        });

        fileBtn?.addEventListener('click', () => fileInput?.click());
        fileInput?.addEventListener('change', () => {
            const file = fileInput.files?.[0];
            fileName.textContent = file ? file.name : '';
            _updateUploadSubmit();
        });
        accountInput?.addEventListener('input', _updateUploadSubmit);

        function _updateUploadSubmit() {
            const hasFile = fileInput?.files?.length > 0;
            const hasLabel = accountInput?.value.trim().length > 0;
            submit.disabled = !(hasFile && hasLabel);
        }

        cancel?.addEventListener('click', _closeUpload);

        submit?.addEventListener('click', async () => {
            const file = fileInput.files[0];
            const label = accountInput.value.trim();
            if (!file || !label) return;
            submit.disabled = true;
            submit.textContent = 'Uploading...';
            status.classList.add('hidden');

            const fd = new FormData();
            fd.append('file', file);
            fd.append('account_label', label);

            try {
                const res = await fetch('/finance/upload_csv', { method: 'POST', body: fd });
                const data = await res.json();
                if (res.ok) {
                    status.textContent = data.message || 'Upload complete';
                    status.className = 'fin-upload-status success';
                    setTimeout(() => { _closeUpload(); _refreshDashboard(); }, 2000);
                } else {
                    status.textContent = data.detail || "Couldn't read that file. Make sure it's a CIBC chequing or credit card CSV export.";
                    status.className = 'fin-upload-status error';
                    submit.disabled = false;
                    submit.textContent = 'Upload';
                }
            } catch (e) {
                status.textContent = 'Upload failed. Check your connection and try again.';
                status.className = 'fin-upload-status error';
                submit.disabled = false;
                submit.textContent = 'Upload';
            }
        });

        function _closeUpload() {
            panel?.classList.add('hidden');
            trigger.style.display = '';
            fileInput.value = '';
            accountInput.value = '';
            fileName.textContent = '';
            submit.disabled = true;
            submit.textContent = 'Upload';
            status.classList.add('hidden');
        }
    }

    // ---- Expenses list renderer ----
    function _renderExpensesList(data) {
        const container = document.getElementById('fin-tx-list');
        const empty = document.getElementById('fin-tx-empty');
        const txns = data.transactions || [];

        if (!txns.length) {
            container.innerHTML = '';
            empty?.classList.remove('hidden');
            return;
        }
        empty?.classList.add('hidden');

        // Group by month (YYYY-MM)
        const groups = {};
        txns.forEach(tx => {
            const ym = tx.date.substring(0, 7); // "YYYY-MM"
            if (!groups[ym]) groups[ym] = { txns: [], debitTotal: 0 };
            groups[ym].txns.push(tx);
            if (tx.type === 'debit') groups[ym].debitTotal += tx.amount;
        });

        // Sort months newest first
        const sortedMonths = Object.keys(groups).sort().reverse();

        let html = '';
        sortedMonths.forEach(ym => {
            const g = groups[ym];
            const [y, m] = ym.split('-');
            const dt = new Date(parseInt(y), parseInt(m) - 1);
            const monthLabel = dt.toLocaleDateString('en-US', { month: 'long', year: 'numeric' });

            html += `<div class="fin-month-group" data-month="${ym}">`;
            html += `<div class="fin-month-group-header" onclick="this.parentElement.classList.toggle('collapsed')">`;
            html += `<span><span class="fin-month-toggle">&#x25BC;</span> <span class="fin-month-name">${monthLabel}</span></span>`;
            html += `<span class="fin-month-total">$${g.debitTotal.toFixed(2)}</span>`;
            html += `</div>`;
            html += `<div class="fin-month-group-body">`;

            g.txns.forEach(tx => {
                // Format date as "Mar 17, 2026"
                const txDate = new Date(tx.date + 'T00:00:00');
                const dateStr = txDate.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });

                const sourceClass = tx.account_type === 'chequing' ? 'fin-tx-source-bank' : 'fin-tx-source-credit';
                const sourceLabel = tx.account_type === 'chequing' ? 'Bank' : 'Credit Card';
                const amountClass = tx.type === 'credit' ? 'fin-tx-amount fin-tx-credit' : 'fin-tx-amount';
                const amountStr = tx.type === 'credit' ? `+$${tx.amount.toFixed(2)}` : `$${tx.amount.toFixed(2)}`;

                html += `<div class="fin-tx-row">`;
                html += `<span class="fin-tx-date">${dateStr}</span>`;
                html += `<span class="fin-tx-desc" title="${tx.description}">${tx.description}</span>`;
                html += `<span class="fin-tx-source ${sourceClass}">${sourceLabel}</span>`;
                html += `<span class="fin-tx-category">${tx.category}</span>`;
                html += `<span class="${amountClass}">${amountStr}</span>`;
                html += `</div>`;
            });

            html += `</div></div>`;
        });

        container.innerHTML = html;
    }

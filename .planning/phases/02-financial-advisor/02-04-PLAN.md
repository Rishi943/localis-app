function _renderLineChart(data) {
    const canvas = document.getElementById('fin-line-chart');
    const emptyOverlay = document.getElementById('fin-line-empty');
    if (!canvas) return;

    const trend = data.trend || []; // [{period: "2026-01", total: 1234.56, count: 45}, ...]

    if (!trend.length) {
        emptyOverlay?.classList.remove('hidden');
        if (_lineChart) { _lineChart.destroy(); _lineChart = null; }
        return;
    }
    emptyOverlay?.classList.add('hidden');

    const labels = trend.map(t => {
        const [y, m] = t.period.split('-');
        const dt = new Date(parseInt(y), parseInt(m) - 1);
        return dt.toLocaleDateString('en-US', { month: 'short', year: 'numeric' });
    });
    const values = trend.map(t => t.total);

    const config = {
        type: 'line',
        data: {
            labels,
            datasets: [{
                data: values,
                borderColor: '#5A8CFF',
                backgroundColor: 'rgba(90, 140, 255, 0.08)',
                fill: true,
                tension: 0.4,
                pointRadius: 3,
                pointBackgroundColor: '#5A8CFF',
                pointBorderColor: '#5A8CFF',
                borderWidth: 2,
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            plugins: {
                legend: { display: false },
                tooltip: {
                    backgroundColor: 'rgba(15,15,15,0.9)',
                    borderColor: 'rgba(255,255,255,0.12)',
                    borderWidth: 1,
                    titleColor: '#fff',
                    bodyColor: 'rgba(255,255,255,0.7)',
                    padding: 8,
                    callbacks: {
                        label: function(ctx) {
                            return '$' + ctx.parsed.y.toLocaleString('en-CA', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
                        }
                    }
                }
            },
            scales: {
                x: {
                    grid: { display: false },
                    ticks: { color: 'rgba(255,255,255,0.4)', font: { size: 11, family: 'Inter' } },
                    border: { color: 'rgba(255,255,255,0.06)' }
                },
                y: {
                    grid: { color: 'rgba(255,255,255,0.06)' },
                    ticks: {
                        color: 'rgba(255,255,255,0.4)',
                        font: { size: 11, family: 'Inter' },
                        callback: function(v) { return '$' + v.toLocaleString(); }
                    },
                    border: { display: false }
                }
            }
        }
    };

    if (_lineChart) {
        _lineChart.data.labels = labels;
        _lineChart.data.datasets[0].data = values;
        _lineChart.update('active');
    } else {
        _lineChart = new Chart(canvas.getContext('2d'), config);
    }
}

function _renderDonutChart(data) {
    const canvas = document.getElementById('fin-donut-chart');
    const emptyOverlay = document.getElementById('fin-donut-empty');
    if (!canvas) return;

    const categories = (data.categories || []).filter(c => c.total > 0);

    if (!categories.length) {
        emptyOverlay?.classList.remove('hidden');
        if (_donutChart) { _donutChart.destroy(); _donutChart = null; }
        return;
    }
    emptyOverlay?.classList.add('hidden');

    // Sort by total descending
    categories.sort((a, b) => b.total - a.total);

    const BLUE_PALETTE = [
        'rgba(90,140,255,1)',
        'rgba(90,140,255,0.85)',
        'rgba(90,140,255,0.7)',
        'rgba(90,140,255,0.55)',
        'rgba(90,140,255,0.4)',
        'rgba(90,140,255,0.28)',
        'rgba(90,140,255,0.18)',
        'rgba(90,140,255,0.10)',
    ];

    const labels = categories.map(c => c.category);
    const values = categories.map(c => c.total);
    const colors = categories.map((_, i) => BLUE_PALETTE[i % BLUE_PALETTE.length]);
    const totalSpend = values.reduce((a, b) => a + b, 0);

    const config = {
        type: 'doughnut',
        data: {
            labels,
            datasets: [{
                data: values,
                backgroundColor: colors,
                borderColor: 'rgba(15,15,15,0.6)',
                borderWidth: 1,
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            cutout: '65%',
            plugins: {
                legend: {
                    position: 'right',
                    labels: {
                        color: 'rgba(255,255,255,0.6)',
                        font: { size: 12, family: 'Inter' },
                        padding: 8,
                        generateLabels: function(chart) {
                            const data = chart.data;
                            return data.labels.map((label, i) => {
                                const val = data.datasets[0].data[i];
                                const pct = totalSpend > 0 ? ((val / totalSpend) * 100).toFixed(0) : 0;
                                return {
                                    text: `${label} ${pct}%`,
                                    fillStyle: data.datasets[0].backgroundColor[i],
                                    strokeStyle: 'transparent',
                                    index: i,
                                    hidden: false,
                                };
                            });
                        }
                    }
                },
                tooltip: {
                    backgroundColor: 'rgba(15,15,15,0.9)',
                    borderColor: 'rgba(255,255,255,0.12)',
                    borderWidth: 1,
                    titleColor: '#fff',
                    bodyColor: 'rgba(255,255,255,0.7)',
                    padding: 8,
                    callbacks: {
                        label: function(ctx) {
                            const pct = totalSpend > 0 ? ((ctx.parsed / totalSpend) * 100).toFixed(1) : 0;
                            return `$${ctx.parsed.toLocaleString('en-CA', {minimumFractionDigits: 2})} (${pct}%)`;
                        }
                    }
                }
            }
        }
    };

    if (_donutChart) {
        _donutChart.data.labels = labels;
        _donutChart.data.datasets[0].data = values;
        _donutChart.data.datasets[0].backgroundColor = colors;
        _donutChart.update('active');
    } else {
        _donutChart = new Chart(canvas.getContext('2d'), config);
    }
}
```

function _renderBudgetSidebar(data) {
    const container = document.getElementById('fin-budget-rows');
    if (!container) return;

    const ALL_CATEGORIES = ['Food', 'Transport', 'Shopping', 'Utilities', 'Entertainment', 'Health & Fitness', 'Government & Fees', 'Other'];
    const categories = data.categories || []; // [{category, total, count}, ...]
    const budgets = data.budgets || {};       // {Food: 500, Transport: 200, ...}

    // Build a map of actual spend by category
    const spendMap = {};
    categories.forEach(c => { spendMap[c.category] = c.total; });

    let html = '';
    ALL_CATEGORIES.forEach(cat => {
        const spent = spendMap[cat] || 0;
        const budget = budgets[cat] || 0;
        const hasBudget = budget > 0;

        html += `<div class="fin-budget-row">`;
        html += `<div class="fin-budget-row-header">`;
        html += `<span class="fin-budget-cat-name">${cat}</span>`;
        html += `<span class="fin-budget-amounts">`;
        html += `<span class="fin-budget-spent">$${spent.toFixed(0)}</span>`;
        if (hasBudget) {
            html += ` <span class="fin-budget-of">/ $${budget.toFixed(0)}</span>`;
        } else {
            html += ` <span class="fin-budget-no-budget">No budget</span>`;
        }
        html += `</span></div>`;

        if (hasBudget) {
            const pct = Math.min((spent / budget) * 100, 100);
            const ratio = spent / budget;
            let barClass = 'fin-budget-bar-fill';
            if (ratio > 1) barClass += ' fin-bar-red';
            else if (ratio >= 0.85) barClass += ' fin-bar-amber';
            // default is accent-primary (blue) via CSS

            html += `<div class="fin-budget-bar-track"><div class="${barClass}" style="--pct: ${pct.toFixed(1)}"></div></div>`;
        } else {
            html += `<div class="fin-budget-bar-ghost"></div>`;
        }

        html += `</div>`;
    });

    container.innerHTML = html;

    // Trigger CSS transition by setting --pct after paint
    requestAnimationFrame(() => {
        container.querySelectorAll('.fin-budget-bar-fill').forEach(el => {
            // Force reflow to trigger transition
            el.offsetHeight;
        });
    });
}
```
